\begin{problem}
	Skrifið forrit sem les inn eina línu sem inniheldur eitt eða fleiri orð. Forritið skrifar svo aftur út línuna, nema með hverju orði öfugsnúnu.

\begin{example}
\exmp{%
\underline{Halló Heimur}
óllaH rumieH
}%
\end{example}
\begin{example}
\exmp{%
\underline{hahahahahah lol}
hahahahahah lol
}%
\end{example}
\begin{example}
\exmp{%
\underline{ninppekranutirrof re emosewa}
forritunarkeppnin er awesome
}%
\end{example}
\end{problem}
