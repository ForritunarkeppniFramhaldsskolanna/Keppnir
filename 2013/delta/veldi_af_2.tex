\begin{problem}
	Skrifið forrit sem spyr um jákvæða heiltölu $n$. Forritið á svo, fyrir hverja tölu $i$ frá $0$ upp í $n$, að skrifa út línu á forminu "`\texttt{2\^{}$i$ = }"' og svo útkomuna úr formúlunni.

\begin{example}
\exmp{%
n = \underline{3}
2\^{}0 = 1
2\^{}1 = 2
2\^{}2 = 4
2\^{}3 = 8
}%
\end{example}
\begin{example}
\exmp{%
n = \underline{8}
2\^{}0 = 1
2\^{}1 = 2
2\^{}2 = 4
2\^{}3 = 8
2\^{}4 = 16
2\^{}5 = 32
2\^{}6 = 64
2\^{}7 = 128
2\^{}8 = 256
}%
\end{example}
\end{problem}
